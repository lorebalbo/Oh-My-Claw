**Create `OhMyClaw/UI/IconAnimator.swift`:**

```swift
import Foundation

/// Manages menu bar icon animation frames during file processing.
/// Cycles through SF Symbol variants at a fixed interval to convey activity.
@MainActor
@Observable
final class IconAnimator {
    private(set) var currentFrame: Int = 0
    private var timer: Timer?

    /// SF Symbol names to cycle through during processing.
    private let frames = [
        "arrow.down.doc",
        "arrow.down.doc.fill",
        "arrow.up.doc.fill"
    ]

    /// The SF Symbol name for the current animation frame.
    var currentIconName: String {
        frames[currentFrame % frames.count]
    }

    /// Start the animation timer. Idempotent — no-op if already running.
    func startAnimating() {
        guard timer == nil else { return }
        currentFrame = 0
        timer = Timer.scheduledTimer(withTimeInterval: 0.5, repeats: true) { [weak self] _ in
            Task { @MainActor in
                self?.currentFrame += 1
            }
        }
    }

    /// Stop the animation and reset to the first frame.
    func stopAnimating() {
        timer?.invalidate()
        timer = nil
        currentFrame = 0
    }
}
```

Key constraints:
- `@MainActor` — all icon state mutations happen on main thread for UI consistency
- `@Observable` — so `OhMyClawApp` label view builder re-renders when `currentFrame` changes
- Timer interval 0.5s (2 FPS) — fast enough to convey activity, slow enough not to be distracting
- 3 frames for a smooth cycle: outline → filled → up-arrow-filled → repeat
- `weak self` in timer closure to avoid retain cycle
- `guard timer == nil` makes `startAnimating()` idempotent

```swift
import SwiftUI

@main
struct OhMyClawApp: App {
    @State private var coordinator = AppCoordinator()

    var body: some Scene {
        MenuBarExtra {
            MenuBarView()
                .environment(coordinator)
        } label: {
            switch coordinator.appState.monitoringState {
            case .paused:
                Image(systemName: coordinator.appState.menuBarIcon)
                    .symbolRenderingMode(.hierarchical)
            case .error:
                Image(systemName: coordinator.appState.menuBarIcon)
            case .processing:
                Image(systemName: coordinator.iconAnimator.currentIconName)
            case .idle:
                Image(systemName: coordinator.appState.menuBarIcon)
            }
        }
        .menuBarExtraStyle(.window)
    }
}
```

Reading `coordinator.iconAnimator.currentIconName` directly in the `.processing` case ensures the label re-renders on every timer tick, because `IconAnimator` is `@Observable` and `currentFrame` changes trigger observation.
