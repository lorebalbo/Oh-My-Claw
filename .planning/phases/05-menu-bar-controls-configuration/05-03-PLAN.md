**Add binding computed properties:**

**Duration binding — auto-saves on change:**
```swift
private var durationBinding: Binding<Int> {
    Binding(
        get: {
            coordinator.configStore?.config.audio.minDurationSeconds ?? 60
        },
        set: { newValue in
            guard var config = coordinator.configStore?.config else { return }
            config.audio.minDurationSeconds = newValue
            coordinator.configStore?.save(config)
        }
    )
}
```

**Quality cutoff binding — auto-saves on change:**

**Port binding — bidirectional with local text state:**
```swift
private var portBinding: Binding<String> {
    Binding(
        get: {
            if portText.isEmpty {
                return String(coordinator.configStore?.config.pdf.lmStudioPort ?? 1234)
            }
            return portText
        },
        set: { newValue in
            portText = newValue
        }
    )
}
```

**Port validation method:**
```swift
private func validateAndSavePort() {
    let currentPort = coordinator.configStore?.config.pdf.lmStudioPort ?? 1234

    guard let port = Int(portText), (1...65535).contains(port) else {
        // Invalid input — reset to current valid value
        portText = String(currentPort)
        return
    }

    guard var config = coordinator.configStore?.config else { return }
    config.pdf.lmStudioPort = port
    coordinator.configStore?.save(config)
    // Clear local state so binding reads from config on next get
    portText = ""
}
```
